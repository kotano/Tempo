@startuml

title "Participants - Sequence Diagram"

actor user
boundary GUI
control list
entity json_file

group alt edit text
    user->GUI:click task
    GUI->list
    activate GUI
    activate list
    list->json_file:search task
    activate json_file
    json_file-->GUI: return task
    GUI-->user
    activate user
    user-> user: edit text
    return 
    GUI->json_file:send edited text
    json_file->json_file: save edited task
    return return edited task
    list-->GUI: show edited task
    deactivate list
    deactivate GUI
    
else write new
    user->GUI:click new task
    GUI->list
    activate GUI
    activate list
    list->list: create a new window
    return return clear task
    GUI-->user
    activate user
    user->user:type task
    return 
    GUI->list:send task
    activate list
    list->json_file:send to save
    activate json_file
    json_file->json_file: save task
    return return saved task
   
    list-->GUI: show new task
    deactivate list
    deactivate GUI
end

@enduml
